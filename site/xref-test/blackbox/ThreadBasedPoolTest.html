126     CountingExpiration expiration = expire(info -> toExpire.contains(info.getPoolable()));

155     }

339   void disregardPileMustNotPreventBackgroundExpirationFromCheckingObjects(Taps taps) throws Exception {
340     if (taps == Taps.THREAD_LOCAL) {
341       return; // It is not safe to use the thread local tap in this test.
342     }
343     CountDownLatch firstThreadReady = new CountDownLatch(1);
344     CountDownLatch firstThreadPause = new CountDownLatch(1);
345     builder.setSize(2);
346     reducedBackgroundExpirationCheckDelay();
347     createPool();
348     PoolTap<GenericPoolable> tap = taps.get(this);
349 
350     // Make the first object TLR claimed in another thread.
351     Future<?> firstThread = forkFuture(() -> {
352       tap.claim(longTimeout).release();
353       GenericPoolable obj = tap.claim(longTimeout);
354       firstThreadReady.countDown();
355       firstThreadPause.await();
356       obj.release();
357       return null;
358     });
359 
360     firstThreadReady.await();
361 
362     // Move the now TLR claimed object to the front of the queue.
363     forkFuture($claimRelease(tap, longTimeout)).get();
364 
365     // Now this claim should move the first object into the disregard pile.
366     GenericPoolable obj = tap.claim(longTimeout);
367     obj.expire(); // Expire the slot, so expiration should pick it up.
368     firstThreadPause.countDown();
369     firstThread.get();
370     allocator.clearLists();
371     obj.release(); // And now we allow the allocator thread to get to it.
372 
373     // By expiring the objects at this point, we should observe that
374     // the 'obj' gets deallocated.
375     boolean found;
376     do {
377       Thread.sleep(10);
378       found = allocator.getDeallocations().contains(obj);
379     } while (!found);

538   /**
539    * Likewise as in {@link #mustStillBeUsableAfterExceptionInAllocate}, a pool
540    * must not break its internal invariants if a Reallocator throws an exception
541    * in reallocate, and it must still be usable after the exception has bubbled out.
542    * We test for this by configuring a Reallocator that always throws on
543    * reallocate, and we also configure an Expiration that will mark the first
544    * slot it checks as expired. Then, when we call claim, that first live slot
545    * will be sent back for reallocation, which will throw and poison the slot.
546    * Then the slot comes back to our still on-going claim, which throws
547    * because of the poison. The slot then gets sent back again, and now,
548    * because of the poison, it will not be reallocated, but instead have a
549    * fresh Poolable allocated anew. This new good Poolable is what we get out
550    * of the last call to claim.
551    */
552   @ParameterizedTest
553   @EnumSource(Taps.class)
554   void mustStillBeUsableAfterExceptionInReallocateWithBackgroundThread(Taps taps) throws Exception {
555     final AtomicBoolean throwInAllocate = new AtomicBoolean();
556     final AtomicBoolean hasExpired = new AtomicBoolean();
557     final CountDownLatch allocationLatch = new CountDownLatch(2);
558     builder.setAllocator(reallocator(
559         alloc($if(throwInAllocate,
560             $throw(new RuntimeException("boo")),
561             $countDown(allocationLatch, $new))),
562         realloc($throw(new RuntimeException("boo")))));
563     builder.setExpiration(expire($expiredIf(hasExpired)));
564     createPool();
565     PoolTap<GenericPoolable> tap = taps.get(this);
566     tap.claim(longTimeout).release(); // object now allocated
567     throwInAllocate.set(true);
568     hasExpired.set(true);
569     try {
570       tap.claim(longTimeout);
571       fail("claim should have thrown");
572     } catch (PoolException ignore) {}
573     throwInAllocate.set(false);
574     hasExpired.set(false);
575     allocationLatch.await();
576     GenericPoolable claim = tap.claim(longTimeout);
577     assertThat(claim).isNotNull();
578     claim.release();
579   }

581   /**
582    * Here's the scenario we're trying to target:
583    *
584    * - You (or your thread) do a successful claim, and triumphantly stores it
585    *   in the ThreadLocal cache.
586    * - You then return the object after use, so now it's back in the
587    *   live-queue for others to grab.
588    * - Someone else tries to claim the object, but decides that it has expired,
589    *   and sends it off through the dead-queue to be reallocated.
590    * - The reallocation fails for some reason, and the slot is now poisoned.
591    * - You want to claim an object again, and start by looking in the
592    *   ThreadLocal cache.
593    * - You find the slot for the object you had last, but the slot is poisoned.
594    * - Now, because you found it in the ThreadLocal cache â and notably did
595    *   *not* pull it off of the live-queue â you cannot just put it on the
596    *   dead-queue, because that could lead to unbounded memory use.
597    * - Instead, it has to be marked as live, and we instead have to wait for
598    *   someone to pull it off of the live-queue, check the poison again, and
599    *   *then* put it on the dead-queue.
600    * - Your ThreadLocal reclaim attempt then end in throwing the poison,
601    *   wrapped in a PoolException.
602    * - Sadly, this process does not involve clearing out the ThreadLocal cache,
603    *   so if you quickly catch the exception and try to claim again, you will
604    *   find the same exact poisoned slot and go through the same routine, that
605    *   ends in a thrown exception and a poisoned slot still left in the
606    *   ThreadLocal cache.
607    */
608   @ParameterizedTest
609   @EnumSource(Taps.class)
610   void mustNotCachePoisonedSlots(Taps taps) throws Exception {
611     // First we prime the possible thread-local cache to a particular object.
612     // Then we instruct the allocator to always throw an exception when it is
613     // told to allocate on that particular slot.
614     // Then, in another thread, we mark all objects in the pool as expired.
615     // Once we have observed a reallocation attempt at our primed slot, we
616     // try to reclaim it. The reclaim must not throw an exception because of
617     // the cached poisoned slot.
618     builder.setSize(1);
619     noBackgroundExpirationChecking();
620 
621     // Enough permits for each initial allocation:
622     final Semaphore semaphore = new Semaphore(1);
623 
624     final AtomicBoolean hasExpired = new AtomicBoolean(false);
625     builder.setExpiration(expire($expiredIf(hasExpired)));
626 
627     final String allocationCause = "allocation blew up!";
628     final AtomicReference<Slot> failOnAllocatingSlot =
629         new AtomicReference<>();
630     final AtomicInteger observedFailedAllocation = new AtomicInteger();
631     Action observeFailure = (slot, obj) -> {
632       if (slot == failOnAllocatingSlot.get()) {
633         observedFailedAllocation.incrementAndGet();
634         failOnAllocatingSlot.set(null);
635         throw new RuntimeException(allocationCause);
636       }
637       return new GenericPoolable(slot);
638     };
639     allocator = allocator(alloc($acquire(semaphore, observeFailure)));

661     semaphore.release(Integer.MAX_VALUE);

672     hasExpired.set(false);
