66    @AfterEach
67    void verifyObjectsAreNeverDeallocatedMoreThanOnce() throws InterruptedException {
68      assertTrue(pool.shutdown().await(shortTimeout), "pool should have been shut down by the test");
69      pool = null;
70  
71      List<GenericPoolable> deallocated = allocator.getDeallocations();
72      // Synchronize to avoid ConcurrentModification with background thread
73      //noinspection SynchronizationOnLocalVariableOrMethodParameter
74      synchronized (deallocated) {
75        deallocated.sort(Comparator.comparingInt(System::identityHashCode));
76        Iterator<GenericPoolable> iter = deallocated.iterator();
77        List<GenericPoolable> duplicates = new ArrayList<>();
78        if (iter.hasNext()) {
79          GenericPoolable a = iter.next();
80          while (iter.hasNext()) {
81            GenericPoolable b = iter.next();
82            if (a == b) {
83              duplicates.add(b);
84            }
85            a = b;
86          }
87        }
88        assertThat(duplicates).isEmpty();
89      }
90      allocator = null;
91    }

98    @Test
99    void highContentionMustNotCausePoolLeakage() throws Exception {
100     createPool();
101 
102     Runnable runner = createTaskClaimReleaseUntilShutdown(pool);
103 
104     Future<?> future = executor.submit(runner);
105     EXECUTOR_EXTENSION.printOnFailure(future);
106 
107     long deadline = System.nanoTime() + TimeUnit.SECONDS.toNanos(5);
108     do {
109       pool.claim(longTimeout).release();
110     } while (System.nanoTime() < deadline);
111     assertTrue(pool.shutdown().await(longTimeout));
112     future.get();
113   }

115   private Runnable createTaskClaimReleaseUntilShutdown(
116       final Pool<GenericPoolable> pool,
117       final Class<? extends Throwable>... acceptableExceptions) {
118     return () -> {
119       for (;;) {
120         try {
121           pool.claim(longTimeout).release();
122         } catch (InterruptedException ignore) {
123           // This is okay
124         } catch (IllegalStateException e) {
125           assertThat(e).hasMessage("Pool has been shut down");
126           break;
127         } catch (PoolException e) {
128           assertThat(e.getCause().getClass()).isIn(asList(acceptableExceptions));
129         }
130       }
131     };
132   }

159     }

206     // The shutdown completes if no objects are leaked
207     assertTrue(pool.shutdown().await(longTimeout));
208 
209     for (Future<?> future : futures) {
210       // Also verify that no unexpected exceptions were thrown
211       future.get();
212     }

267     CountDownLatch startLatch = new CountDownLatch(startingSize);
268     Semaphore semaphore = new Semaphore(0);
269     allocator = allocator(
270         alloc($countDown(startLatch, $new)),
271         dealloc($release(semaphore, $null)));
272     builder.setSize(startingSize).setBackgroundExpirationCheckDelay(10);
273     builder.setAllocator(allocator);
274     createPool();
275     startLatch.await();

278       objs.add(pool.claim(longTimeout));

334   @Test
335   void explicitlyExpiredSlotsMustNotCauseBackgroundCPUBurn()
336       throws InterruptedException {
337     final ThreadMXBean threads = ManagementFactory.getThreadMXBean();
338     final AtomicLong lastUserTimeIncrement = new AtomicLong();
339     assumeTrue(threads.isCurrentThreadCpuTimeSupported());
340     allocator = allocator(alloc(
341         measureLastCPUTime(threads, lastUserTimeIncrement)));
342     builder.setAllocator(allocator);
343     builder.setSize(2);
344     createPool();
345     GenericPoolable a = pool.claim(longTimeout);
346     GenericPoolable b = pool.claim(longTimeout);
347     a.expire();
348     Thread.sleep(10);
349     a.release();
350     a = pool.claim(longTimeout);
351     long millisecondsAllowedToBurnCPU = 5000;
352     Thread.sleep(millisecondsAllowedToBurnCPU);
353     b.expire();
354     b.release();
355     b = pool.claim(longTimeout);
356     a.release();
357     b.release();
358     long millisecondsSpentBurningCPU =
359         TimeUnit.NANOSECONDS.toMillis(lastUserTimeIncrement.get());
360 
361     assertThat(millisecondsSpentBurningCPU).isLessThan(millisecondsAllowedToBurnCPU / 2);
362   }

382   void explicitlyExpiredSlotsThatAreDeallocatedThroughPoolShrinkingMustNotCauseBackgroundCPUBurn()
383       throws InterruptedException {
384     final ThreadMXBean threads = ManagementFactory.getThreadMXBean();
385     final AtomicLong lastUserTimeIncrement = new AtomicLong();
386     final AtomicLong maxUserTimeIncrement = new AtomicLong();
387     assumeTrue(threads.isCurrentThreadCpuTimeSupported());
388     allocator = allocator(alloc(new Action() {
389       boolean first = true;
390 
391       @Override
392       public GenericPoolableenericPoolable.html#GenericPoolable">GenericPoolable apply(Slot slot, GenericPoolable obj) throws Exception {
393         if (first) {
394           threads.setThreadCpuTimeEnabled(true);
395           first = false;
396         }
397         long userTime = threads.getCurrentThreadUserTime();
398         long delta = userTime - lastUserTimeIncrement.get();
399         lastUserTimeIncrement.set(delta);
400         long existingDelta;
401         do {
402           existingDelta = maxUserTimeIncrement.get();
403         } while ( !maxUserTimeIncrement.compareAndSet(
404             existingDelta, Math.max(delta, existingDelta)));
405         return $new.apply(slot, obj);
406       }
407     }));
408     builder.setAllocator(allocator);
409     int size = 30;
410     builder.setSize(size);
411     createPool();
412     LinkedList<GenericPoolable> objs = new LinkedList<>();
413     for (int i = 0; i < size; i++) {
414       GenericPoolable obj = pool.claim(longTimeout);
415       objs.offer(obj);
416       obj.expire();
417     }
418     int newSize = size / 3;
419     pool.setTargetSize(newSize);
420     Iterator<GenericPoolable> itr = objs.iterator();
421     for (int i = size; i >= newSize; i--) {
422       itr.next().release();
423       itr.remove();
424     }
425     long millisecondsAllowedToBurnCPU = 5000;
426     Thread.sleep(millisecondsAllowedToBurnCPU);
427     while (itr.hasNext()) {
428       itr.next().release();
429     }
430     pool.claim(longTimeout).release();
431 
432     long millisecondsSpentBurningCPU =
433         TimeUnit.NANOSECONDS.toMillis(maxUserTimeIncrement.get());
434 
435     assertThat(millisecondsSpentBurningCPU).isLessThan(millisecondsAllowedToBurnCPU / 2);
436   }
