64    @AfterEach
65    void verifyObjectsAreNeverDeallocatedMoreThanOnce() throws InterruptedException {
66      assertTrue(pool.shutdown().await(shortTimeout), "pool should have been shut down by the test");
67      pool = null;
68  
69      List<GenericPoolable> deallocated = allocator.getDeallocations();
70      // Synchronize to avoid ConcurrentModification with background thread
71      //noinspection SynchronizationOnLocalVariableOrMethodParameter
72      synchronized (deallocated) {
73        deallocated.sort(Comparator.comparingInt(System::identityHashCode));
74        Iterator<GenericPoolable> iter = deallocated.iterator();
75        List<GenericPoolable> duplicates = new ArrayList<>();
76        if (iter.hasNext()) {
77          GenericPoolable a = iter.next();
78          while (iter.hasNext()) {
79            GenericPoolable b = iter.next();
80            if (a == b) {
81              duplicates.add(b);
82            }
83            a = b;
84          }
85        }
86        assertThat(duplicates).isEmpty();
87      }
88      allocator = null;
89    }

95    @org.junit.jupiter.api.Timeout(160)
96    @Test
97    void highContentionMustNotCausePoolLeakage() throws Exception {
98      createPool();
99  
100     Runnable runner = createTaskClaimReleaseUntilShutdown(pool);
101 
102     Future<?> future = executor.submit(runner);
103     executorExtension.printOnFailure(future);
104 
105     long deadline = System.nanoTime() + TimeUnit.SECONDS.toNanos(5);
106     do {
107       pool.claim(longTimeout).release();
108     } while (System.nanoTime() < deadline);
109     assertTrue(pool.shutdown().await(longTimeout));
110     future.get();
111   }

113   private Runnable createTaskClaimReleaseUntilShutdown(
114       final Pool<GenericPoolable> pool,
115       final Class<? extends Throwable>... acceptableExceptions) {
116     return () -> {
117       for (;;) {
118         try {
119           GenericPoolable obj = pool.claim(longTimeout);
120           obj.release();
121         } catch (InterruptedException ignore) {
122           // This is okay
123         } catch (IllegalStateException e) {
124           assertThat(e).hasMessage("Pool has been shut down");
125           break;
126         } catch (PoolException e) {
127           assertThat(e.getCause().getClass()).isIn(asList(acceptableExceptions));
128         }
129       }
130     };
131   }

167 
168       @Override
169       public GenericPoolableenericPoolable.html#GenericPoolable">GenericPoolable apply(Slot slot, GenericPoolable obj) throws Exception {
170         // About 20% of allocations, deallocations and reallocations will throw
171         if (rnd.nextInt(1024) < 201) {
172           throw new SomeRandomException();
173         }
174         return new GenericPoolable(slot);
175       }
176     };
177     allocator = reallocator(
178         alloc(fallibleAction),
179         dealloc(fallibleAction),
180         realloc(fallibleAction));
181     builder.setAllocator(allocator);
182     builder.setExpiration(info -> {
183       int x = ThreadLocalRandom.current().nextInt();
184       if ((x & 0xFF) > 250) {
185         // About 3% of checks throw an exception
186         throw new SomeRandomException();
187       }
188       // About 1 in 8 checks causes expiration
189       return (x & 0x0F) < 0x02;
190     });
191 
192     createPool();
193 
194     List<Future<?>> futures = new ArrayList<>();
195     for (int i = 0; i < 64; i++) {
196       Runnable runner = createTaskClaimReleaseUntilShutdown(
197           pool,

199       futures.add(executor.submit(runner));
200     }
201     executorExtension.printOnFailure(futures);
202

204 
205     // The shutdown completes if no objects are leaked
206     assertTrue(pool.shutdown().await(longTimeout));
207 
208     for (Future<?> future : futures) {
209       // Also verify that no unexpected exceptions were thrown
210       future.get();
211     }

214   @org.junit.jupiter.api.Timeout(160)
215   @Test
216   void mustNotDeallocateNullsFromLiveQueueDuringShutdown() throws Exception {
217     int startingSize = 256;
218     CountDownLatch startLatch = new CountDownLatch(startingSize);
219     Semaphore semaphore = new Semaphore(0);
220     allocator = allocator(
221         alloc($countDown(startLatch, $new)),
222         dealloc($release(semaphore, $null)));
223     builder.setSize(startingSize);
224     builder.setAllocator(allocator);
225     createPool();
226     startLatch.await();
227     List<GenericPoolable> objs = new ArrayList<>();
228     for (int i = 0; i < startingSize; i++) {
229       objs.add(pool.claim(longTimeout));
230     }
231 
232     Completion completion = pool.shutdown();
233     Future<Boolean> completionFuture = executor.submit(() -> completion.await(longTimeout));
234     int size = startingSize;
235     List<GenericPoolable> subList = objs.subList(0, startingSize - 1);
236     for (GenericPoolable obj : subList) {
237       size--;
238       // It's important that the wait mask produces values greater than the
239       // allocation threads idle wait time.
240       assertFalse(semaphore.tryAcquire(size & 127, TimeUnit.MILLISECONDS));
241       obj.release();
242       semaphore.acquire();
243     }
244 
245     assertThat(allocator.getDeallocations()).isEqualTo(subList);
246 
247     objs.get(startingSize - 1).release();
248     assertTrue(completionFuture.get(1, TimeUnit.MINUTES), "shutdown timeout elapsed");
249   }

252   void explicitlyExpiredSlotsMustNotCauseBackgroundCPUBurn()
253       throws InterruptedException {
254     final ThreadMXBean threads = ManagementFactory.getThreadMXBean();
255     final AtomicLong lastUserTimeIncrement = new AtomicLong();
256     assumeTrue(threads.isCurrentThreadCpuTimeSupported());
257     allocator = allocator(alloc(
258         measureLastCPUTime(threads, lastUserTimeIncrement)));
259     builder.setAllocator(allocator);
260     builder.setSize(2);
261     createPool();
262     GenericPoolable a = pool.claim(longTimeout);
263     GenericPoolable b = pool.claim(longTimeout);
264     a.expire();
265     Thread.sleep(10);
266     a.release();
267     a = pool.claim(longTimeout);
268     long millisecondsAllowedToBurnCPU = 5000;
269     Thread.sleep(millisecondsAllowedToBurnCPU);
270     b.expire();
271     b.release();
272     b = pool.claim(longTimeout);
273     a.release();
274     b.release();
275     long millisecondsSpentBurningCPU =
276         TimeUnit.NANOSECONDS.toMillis(lastUserTimeIncrement.get());
277 
278     assertThat(millisecondsSpentBurningCPU).isLessThan(millisecondsAllowedToBurnCPU / 2);
279   }

282     return new Action() {
283       boolean first = true;
284 
285       @Override
286       public GenericPoolableenericPoolable.html#GenericPoolable">GenericPoolable apply(Slot slot, GenericPoolable obj) throws Exception {

289           first = false;

292           poolable = $new.apply(slot, obj);
293 
294           threads.setThreadCpuTimeEnabled(true);
295           long currentThreadUserTime = threads.getCurrentThreadUserTime();
296           cpuTimeSum.set(currentThreadUserTime);
297         } else {
298           long userTime = threads.getCurrentThreadUserTime();
299           cpuTimeSum.set(userTime - cpuTimeSum.get());

301         }
302         return poolable;

307   @Test
308   void explicitlyExpiredSlotsThatAreDeallocatedThroughPoolShrinkingMustNotCauseBackgroundCPUBurn()
309       throws InterruptedException {
310     final ThreadMXBean threads = ManagementFactory.getThreadMXBean();
311     final AtomicLong lastUserTimeIncrement = new AtomicLong();
312     final AtomicLong maxUserTimeIncrement = new AtomicLong();
313     assumeTrue(threads.isCurrentThreadCpuTimeSupported());
314     allocator = allocator(alloc(new Action() {
315       boolean first = true;
316 
317       @Override
318       public GenericPoolableenericPoolable.html#GenericPoolable">GenericPoolable apply(Slot slot, GenericPoolable obj) throws Exception {
319         if (first) {
320           first = false;
321           GenericPoolable poolable = $new.apply(slot, obj);
322           threads.setThreadCpuTimeEnabled(true);
323           lastUserTimeIncrement.set(threads.getCurrentThreadUserTime());
324           return poolable;
325         }
326         long userTime = threads.getCurrentThreadUserTime();
327         long delta = userTime - lastUserTimeIncrement.get();
328         lastUserTimeIncrement.set(delta);
329         long existingDelta;
330         do {
331           existingDelta = maxUserTimeIncrement.get();
332         } while (!maxUserTimeIncrement.compareAndSet(
333             existingDelta, Math.max(delta, existingDelta)));
334         return $new.apply(slot, obj);
335       }
336     }));
337     builder.setAllocator(allocator);
338     int size = 30;
339     builder.setSize(size);
340     createPool();
341     LinkedList<GenericPoolable> objs = new LinkedList<>();
342     for (int i = 0; i < size; i++) {
343       GenericPoolable obj = pool.claim(longTimeout);
344       objs.offer(obj);
345       obj.expire();
346     }
347     int newSize = size / 3;
348     pool.setTargetSize(newSize);
349     Iterator<GenericPoolable> itr = objs.iterator();
350     for (int i = size; i >= newSize; i--) {
351       itr.next().release();
352       itr.remove();
353     }
354     long millisecondsAllowedToBurnCPU = 5000;
355     Thread.sleep(millisecondsAllowedToBurnCPU);
356     while (itr.hasNext()) {
357       itr.next().release();
358     }
359     pool.claim(longTimeout).release();
360 
361     long millisecondsSpentBurningCPU =
362         TimeUnit.NANOSECONDS.toMillis(maxUserTimeIncrement.get());
363 
364     assertThat(millisecondsSpentBurningCPU).isLessThan(millisecondsAllowedToBurnCPU / 2);
365   }

427     AtomicLong counter = new AtomicLong();

442       long delta = curr - prev;
443       prev = curr;
444       if (i > 40) {
445         assertThat(delta).isLessThanOrEqualTo(5);
446       }
447     }
