133   @Test
134   void allSetterMethodsMustReturnTheSameBuilderInstance() throws Exception {
135     Method[] methods = PoolBuilder.class.getDeclaredMethods();
136     List<Method> setterMethods = new ArrayList<>();
137     for (Method method : methods) {
138       if (method.getName().startsWith("set")) {
139         setterMethods.add(method);
140       }
141     }
142 
143     for (Method setter : setterMethods) {
144       Class<?> parameterType = setter.getParameterTypes()[0];
145       Object arg =
146           parameterType == Boolean.TYPE? true :
147           parameterType == Integer.TYPE? 1 :
148           parameterType == Allocator.class ? allocator() : null;
149       Object result = setter.invoke(builder, arg);
150       assertThat(result).as("return value of setter " + setter).isSameAs(builder);
151     }
152   }
153 
154   @Test
155   void allPublicDeclaredMethodsMustBeSynchronized() {
156     // We don't care about non-overridden public methods of the super-class
157     // (Object) because they don't operate on the state of the PoolBuilder object
158     // anyway.
159     Method[] methods = PoolBuilder.class.getDeclaredMethods();
160     for (Method method : methods) {
161       int modifiers = method.getModifiers();
162       int IS_SYNTHETIC = 0x00001000;
163       if (Modifier.isPublic(modifiers) && (modifiers & IS_SYNTHETIC) == 0) {
164         // That is, this method is both public AND NOT synthetic.
165         // We have to exclude synthetic methods because javac generates one for
166         // bridging the covariant override of clone().
167         assertTrue(Modifier.isSynchronized(modifiers), "Method '" + method + "' should be synchronized.");
168       }
169     }
170   }
171 
172   @Test
173   void builderMustBeCloneable() {
174     CountingAllocator allocator = AlloKit.allocator();
175     ExpireKit.CountingExpiration expiration = ExpireKit.expire();
176     MetricsRecorder metricsRecorder = new LastSampleMetricsRecorder();
177     ThreadFactory factory = r -> null;
178 
179     builder.setExpiration(expiration);
180     builder.setAllocator(allocator);
181     builder.setBackgroundExpirationEnabled(false);
182     builder.setBackgroundExpirationCheckDelay(1313);
183     builder.setMetricsRecorder(metricsRecorder);
184     builder.setPreciseLeakDetectionEnabled(false);
185     builder.setSize(42);
186     builder.setThreadFactory(factory);
187 
188     PoolBuilder<GenericPoolable> clone = builder.clone();
189 
190     assertThat(clone.getExpiration()).isSameAs(expiration);
191     assertThat(clone.getAllocator()).isSameAs(allocator);
192     assertFalse(clone.isBackgroundExpirationEnabled());
193     assertThat(clone.getBackgroundExpirationCheckDelay()).isEqualTo(1313);
194     assertThat(clone.getMetricsRecorder()).isSameAs(metricsRecorder);
195     assertFalse(clone.isPreciseLeakDetectionEnabled());
196     assertThat(clone.getSize()).isEqualTo(42);
197     assertThat(clone.getThreadFactory()).isSameAs(factory);
198   }
