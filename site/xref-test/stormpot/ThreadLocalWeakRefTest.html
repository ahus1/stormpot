16  package stormpot;
17  
18  import org.junit.jupiter.api.Disabled;
19  import org.junit.jupiter.api.Test;
20  
21  import java.lang.ref.WeakReference;
22  
23  import static org.assertj.core.api.Assertions.assertThat;
24  
25  @SuppressWarnings("UnusedAssignment")
26  class ThreadLocalWeakRefTest {
27    /**
28     * Many Java web servers and containers go out of their way to find and null
29     * out ThreadLocals that an application might have created. Question is, is
30     * that really necessary?
31     */
32    @Disabled("This is just for demonstration purpose.")
33    @Test
34    void threadsWeaklyReferenceThreadLocalValues() {
35      // Hypothesis: When we loose the reference to a ThreadLocal (e.g. by a
36      // web app being redeployed and getting a soft restart), and no other
37      // references to that object exists, then it should eventually be garbage
38      // collected.
39  
40      Object obj = new Object();
41      ThreadLocal<Object> threadLocal = new ThreadLocal<>();
42      WeakReference<Object> weakReference = new WeakReference<>(obj);
43      threadLocal.set(obj);
44  
45      // The stack of every thread is a GC-root. We have a strong reference to
46      // obj through the variable on our stack. When we clear it out, the only
47      // strong reference that will be left, will be through the threadLocal.
48      obj = null;
49  
50      // When we loose the reference to the threadLocal, then the values it
51      // referred to will no longer be strongly referenced by our thread.
52      threadLocal = null;
53  
54      // Our obj is now in principle garbage, though we still have a
55      // weakReference to it. The GC does not yet know this, because only the
56      // threadLocal object itself is now weakly referenced. We clear out that
57      // weak reference with a run of the GC.
58      System.gc();
59  
60      // The entry in our ThreadLocalMap is now stale. It still exists, and
61      // strongly references our obj, but it will be cleared next time our
62      // ThreadLocalMap does its clean-up procedure. We cannot invoke the
63      // clean-up procedure directly, but sufficient perturbation of the
64      // ThreadLocalMap will bring it about.
65      ThreadLocal<Object> a = new ThreadLocal<>();
66      a.set(new Object());
67      ThreadLocal<Object> b = new ThreadLocal<>();
68      b.set(new Object());
69  
70      // The entry has now been removed, and our obj is now weakly referenced
71      // by our weakReference above. Invoking another round of GC will clear it.
72      System.gc();
73  
74      // The obj has now been collected because all references are gone.
75      // We observe this by the weakReference being null.
76      assertThat(weakReference.get()).isNull();
77  
78      // This test passes on Java 6, 7 and 8, with default GC settings.
79      // I haven't tried alternative GC settings, but you are welcome to do that.
80    }
81  }
