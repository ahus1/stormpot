16  package stormpot;
17  
18  import org.junit.jupiter.api.Test;
19  import org.openjdk.jol.info.ClassLayout;
20  
21  import java.util.concurrent.TimeUnit;
22  
23  import static org.assertj.core.api.Assertions.assertThat;
24  
25  class BSlotTest {
26    // Blindly assume that cache lines are always 64 bytes wide.
27    // Architectures where this is not the case, will just have to suffer.
28    private static final int CACHE_LINE_SIZE = 64;
29  
30    @Test
31    void slotObjectsShouldBeCacheLineAligned() throws Exception {
32      String arch = System.getProperty("os.arch");
33      if (arch.equals("x86_64") || arch.equals("amd64")) { // Only enable this on 64-bit machines.
34        Pool<Pooled<String>> pool = Pool.of("a");
35        Pooled<String> object = pool.claim(new Timeout(1, TimeUnit.SECONDS));
36        Slot slot = object.slot;
37        ClassLayout classLayout = ClassLayout.parseInstance(slot);
38        long size = classLayout.instanceSize();
39        String description = "BSlot instance size should be an even multiple of 64, " +
40            "but was " + size + ", which is " + (size % CACHE_LINE_SIZE) + " bytes too big:" +
41            System.lineSeparator() + classLayout.toPrintable();
42        assertThat(size % CACHE_LINE_SIZE).as(description).isEqualTo(0);
43      }
44    }
45  
46    @Test
47    void toStringForBSlot() {
48      BSlot<Pooled<String>> slot = new BSlot<>(null, null);
49      assertThat(slot.toString()).isEqualTo("BSolt[DEAD, obj = null, poison = null]");
50      slot.obj = new Pooled<>(slot, "poke");
51      slot.dead2live();
52      assertThat(slot.toString()).isEqualTo("BSolt[LIVING, obj = Pooled[poke], poison = null]");
53      slot.live2claimTlr();
54      assertThat(slot.toString()).isEqualTo("BSolt[TLR_CLAIMED, obj = Pooled[poke], poison = null]");
55      slot.claimTlr2live();
56      slot.live2claim();
57      assertThat(slot.toString()).isEqualTo("BSolt[CLAIMED, obj = Pooled[poke], poison = null]");
58      slot.claim2dead();
59      slot.obj = null;
60      slot.poison = new Exception("boo");
61      assertThat(slot.toString()).isEqualTo(
62          "BSolt[DEAD, obj = null, poison = java.lang.Exception: boo]");
63    }
64  }
