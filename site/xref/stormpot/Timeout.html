16  package stormpot;
17  
18  import java.util.concurrent.TimeUnit;
19  
20  /**
21   * A Timeout represents the maximum amount of time a caller is willing to wait
22   * for a blocking operation to complete.
23   *
24   * Timeouts are independent of their units, so two timeouts of equivalent
25   * duration but constructed in different units, will be equal to each other and
26   * work exactly the same.
27   *
28   * Timeouts are also independent of "calendar time" in the sense that they
29   * represent and work as a duration of absolute time. In other words, timeouts
30   * do not grow or shrink with passing leap seconds or daylight savings time
31   * adjustments.
32   *
33   * @author Chris Vest <mr.chrisvest@gmail.com>
34   */
35  public final class Timeout {
36    private final long timeout;
37    private final TimeUnit unit;
38    private final long timeoutBase;
39    
40    /**
41     * Construct a new timeout with the given value and unit. The unit cannot be
42     * `null`, but the timeout value is unrestricted. The meaning of a
43     * negative timeout value is specific to the implementation of the use site,
44     * but typically means that no amount of blocking or waiting is tolerated.
45     * @param timeout A numerical value for the timeout. Can be zero or negative,
46     * though the meaning is implementation specific.
47     * @param unit The unit of the timeout value. Never `null`.
48     */
49    public Timeout(long timeout, TimeUnit unit) {
50      if (unit == null) {
51        throw new IllegalArgumentException("The TimeUnit cannot be null");
52      }
53      this.timeout = timeout;
54      this.unit = unit;
55      this.timeoutBase = getBaseUnit().convert(timeout, unit);
56    }
57  
58    /**
59     * Get the timeout value in terms of the {@link #getUnit() unit}.
60     * @return A numerical value of the timeout. Possibly zero or negative.
61     */
62    public long getTimeout() {
63      return timeout;
64    }
65  
66    /**
67     * Get the unit for the {@link #getTimeout() timeout value}.
68     * @return The {@link TimeUnit} of the timeout value. Never `null`.
69     */
70    public TimeUnit getUnit() {
71      return unit;
72    }
73  
74    /**
75     * Calculate a deadline, as an instant in the future, in terms of the
76     * {@link #getBaseUnit() base unit}. Once you have a deadline, you can ask
77     * how much time is left until it transpires, with the
78     * {@link #getTimeLeft(long)} method, giving the deadline as an argument.
79     *
80     * If the {@link #getTimeout() timeout value} is really small, zero or
81     * negative, then the deadline might be an instant in the past.
82     * @return A numerical value that represents the deadline from "now" until
83     * this timeout has passed.
84     */
85    @SuppressWarnings("WeakerAccess")
86    public long getDeadline() {
87      return now() + timeoutBase;
88    }
89  
90    /**
91     * Get the timeout value in terms of the {@link #getBaseUnit() base unit}.
92     * @return A numerical value of the timeout. Possibly zero or negative.
93     */
94    public long getTimeoutInBaseUnit() {
95      return timeoutBase;
96    }
97  
98    /**
99     * Calculate the amount of time that is left until the deadline, in terms
100    * of the {@link #getBaseUnit() base unit}. The argument is a deadline value
101    * that has been calculated with the {@link #getDeadline()} method.
102    * @param deadline The reference deadline from {@link #getDeadline()}.
103    * @return The amount of time, in terms of the
104    * {@link #getBaseUnit() base unit}, that is left until the deadline. If
105    * this value is negative, then the deadline has transpired.
106    */
107   @SuppressWarnings("WeakerAccess")
108   public long getTimeLeft(long deadline) {
109     return deadline - now();
110   }
111 
112   /**
113    * Get the unit of precision for the underlying clock, that is used by
114    * {@link #getDeadline()} and {@link #getTimeLeft(long)}.
115    * @return TimeUnit The unit of precision used by the clock in this Timeout.
116    */
117   public TimeUnit getBaseUnit() {
118     return TimeUnit.NANOSECONDS;
119   }
120 
121   private long now() {
122     return System.nanoTime();
123   }
124 
125   @Override
126   public int hashCode() {
127     return 31 * (1 + (int) (timeoutBase ^ (timeoutBase >>> 32)));
128   }
129 
130   /**
131    * Timeouts of equivalent duration are equal, even if they were constructed
132    * with different units.
133    * @return `true` if this Timeout value is equal to the given
134    * Timeout value, `false` otherwise.
135    */
136   @Override
137   public boolean equals(Object obj) {
138     if (!(obj instanceof Timeout)) {
139       return false;
140     }
