16  package stormpot;
17  
18  /**
19   * Reallocators are a special kind of {@link Allocator} that can reallocate
20   * {@link Poolable}s that have expired. This is useful since the {@link Pool}
21   * will typically keep the Poolables around for a relatively long time, with
22   * respect to garbage collection. Because of this, there is a high chance that
23   * the Poolables tenure to the old generation during their life time. This
24   * way, pools often cause a slow, but steady accretion of old generation
25   * garbage, ultimately helping to increase the frequency of expensive full
26   * collections.
27   *
28   * The accretion of old generation garbage is inevitable, but the rate can be
29   * slowed by reusing as much as possible of the Poolable instances, when they
30   * are to be reallocated. This interface is only here to enable this
31   * optimisation, and implementing it is completely optional.
32   *
33   * @author Chris Vest <mr.chrisvest@gmail.com>
34   *
35   * @param <T> any type that implements Poolable.
36   * @since 2.2
37   */
38  public interface Reallocator<T extends Poolable> extends Allocator<T> {
39    /**
40     * Possibly reallocate the given instance of T for the given slot, and
41     * return it if the reallocation was successful, or a fresh replacement if
42     * the instance could not be reallocated.
43     *
44     * This method is effectively equivalent to the following:
45     *
46     * [source,java]
47     * --
48     * include::src/test/java/examples/Examples.java[tag=reallocatorExample]
49     * --
50     *
51     * With the only difference that it may, if possible, reuse the given
52     * expired Poolable, either wholly or in part.
53     *
54     * The state stored in the {@link stormpot.SlotInfo} for the object is reset
55     * upon reallocation, just like it would be in the case of a normal
56     * deallocation-allocation cycle.
57     *
58     * Exceptions thrown by this method may propagate out through the
59     * {@link Pool#claim(Timeout) claim} method of a pool, in the form of being
60     * wrapped inside a {@link PoolException}. Pools must be able to handle these
61     * exceptions in a sane manner, and are guaranteed to return to a working
62     * state if a Reallocator stops throwing exceptions from its reallocate
63     * method.
64     *
65     * Be aware that if the reallocation of an object fails with an exception,
66     * then no attempts will be made to explicitly deallocate that object. This
67     * way, a failed reallocation is implicitly understood to effectively be a
68     * successful deallocation.
69     * @param slot The slot the pool wish to allocate an object for.
70     * Implementers do not need to concern themselves with the details of a
71     * pools slot objects. They just have to call release on them as the
72     * protocol demands.
73     * @param poolable The non-null Poolable instance to be reallocated.
74     * @return A fresh or rejuvenated instance of T. Never `null`.
75     * @throws Exception If the allocation fails.
76     * @see #allocate(Slot)
77     * @see #deallocate(Poolable)
78     */
79    T reallocate(Slot slot, T poolable) throws Exception;
80  }
