16  package stormpot;
17  
18  /**
19   * A Pool is a self-renewable set of objects from which one can claim exclusive
20   * access to elements, until they are released back into the pool.
21   *
22   * Pools are thread-safe, and their {@link #claim(Timeout)} methods can be
23   * called concurrently from multiple threads.
24   * This is a strictly stronger guarantee than {@link PoolTap} provides.
25   *
26   * Pools extend the {@link PoolTap} class, which provides the API for accessing
27   * the objects contained in the pool.
28   *
29   * Pools contain {@link Poolable} objects. When you claim an object in a pool,
30   * you also take upon yourself the responsibility of eventually
31   * {@link Poolable#release() releasing} that object again. By far the most
32   * common idiom to achieve this is with a `try-finally` clause:
33   *
34   * [source,java]
35   * ----
36   * include::../src/test/java/examples/Examples.java[tag=poolClaimExample]
37   * ----
38   *
39   * The pools are resizable, and can have their capacity changed at any time
40   * after they have been created.
41   *
42   * All pools are configured with a certain size, the number of objects that the
43   * pool will have allocated at any one time, and they can change this number on
44   * the fly using the {@link #setTargetSize(int)} method. The change does not
45   * take effect immediately, but instead moves the "goal post" and leaves the
46   * pool to move towards it at its own pace.
47   *
48   * No guarantees can be made about when the pool will actually reach the target
49   * size, because it might depend on how long it takes for a certain number of
50   * objects to be released back into the pool.
51   *
52   * [NOTE]
53   * --
54   * Pools created with {@link Pool#of(Object[])} are _not_ resizable, and calling
55   * {@link Pool#setTargetSize(int)} or {@link ManagedPool#setTargetSize(int)} will
56   * cause an {@link UnsupportedOperationException} to be thrown.
57   * --
58   *
59   * Pools are themselves lifecycled, and should be {@link #shutdown() shut down}
60   * when they are no longer needed.
61   *
62   * Note that Pools are not guaranteed to have overwritten the
63   * {@code Object#finalize()} method. Pools are expected to rely on explicit
64   * clean-up for releasing their resources.
65   *
66   * @author Chris Vest <mr.chrisvest@gmail.com>
67   * @param <T> the type of {@link Poolable} contained in the pool, as determined
68   * by the {@linkplain Pool#from(Allocator) configured allocator}.
69   * @see stormpot.PoolTap
70   */
71  public abstract class Pool<T extends Poolable> extends PoolTap<T> {
72    Pool() {
73    }
74  
75    /**
76     * Get a {@link PoolBuilder} based on the given {@link Allocator} or
77     * {@link Reallocator}, which can then in turn be used to
78     * {@linkplain PoolBuilder#build build} a {@link Pool} instance with the
79     * desired configuration.
80     *
81     * @param allocator The allocator we want our pools to use. This cannot be
82     * `null`.
83     * @param <T> The type of {@link Poolable} that is created by the allocator,
84     * and the type of objects that the configured pools will contain.
85     * @return A {@link PoolBuilder} that admits additional configurations,
86     * before the pool instance is {@linkplain PoolBuilder#build() built}.
87     */
88    public static <T extends Poolable> PoolBuilder<T> from(Allocator<T> allocator) {
89      return new PoolBuilder<>(allocator);
90    }
91  
92    /**
93     * Build a {@link Pool} instance that pools the given set of objects.
94     *
95     * The objects in the pool are never expired, and never deallocated.
96     * Explicitly expired objects simply return to the pool.
97     *
98     * This means that the returned pool has no background allocation thread,
99     * and has no expiration checking overhead when claiming objects.
100    *
101    * The given objects are wrapped in {@link Pooled} objects, which
102    * implement the {@link Poolable} interface.
103    *
104    * Shutting down the returned pool will not cause the given objects
105    * to be deallocated.
106    * The shut down process will complete as soon as the last claimed
107    * object is released back to the pool.
108    *
109    * [NOTE]
110    * --
111    * Passing duplicate objects to this method is not supported.
112    * The behaviour of the pool is unspecified if there are duplicates among
113    * the objects passed as arguments to this method.
114    * --
115    *
116    * @param objects The objects the pool should contain.
117    * @param <T> The type of objects being pooled.
118    * @return A pool of the given objects.
119    */
120   @SafeVarargs
121   public static <T> Pool<Pooled<T>> of(T... objects) {
122     Allocator<Pooled<T>> allocator = new Allocator<>() {
123       private int index;
124       @Override
125       public Pooled<T> allocate(Slot slot) {
126         return new Pooled<>(slot, objects[index++]);
127       }
128 
129       @Override
130       public void deallocate(Pooled<T> poolable) {
131       }
132     };
133     PoolBuilder<Pooled<T>> builder = new PoolBuilder<>(allocator);
134     builder.setSize(objects.length);
135     builder.setPreciseLeakDetectionEnabled(false);
136     builder.setBackgroundExpirationEnabled(false);
137     builder.setExpiration(Expiration.never());
138     return new BlazePool<>(builder, AllocatorProcessFactory.DIRECT);
139   }
140 
141   /**
142    * Initiate the shut down process on this pool, and return a
143    * {@link Completion} instance representing the shut down procedure.
144    *
145    * The shut down process is asynchronous, and the shutdown method is
146    * guaranteed to not wait for any claimed {@link Poolable Poolables} to
147    * be released.
148    *
149    * The shut down process cannot complete before all Poolables are released
150    * back into the pool and {@link Allocator#deallocate(Poolable) deallocated},
151    * and all internal resources (such as threads, for instance) have been
152    * released as well. Only when all of these things have been taken care of,
153    * does the await methods of the Completion return.
154    *
155    * Once the shut down process has been initiated, that is, as soon as this
156    * method is called, the pool can no longer be used and all calls to
157    * {@link #claim(Timeout)} will throw an {@link IllegalStateException}.
158    * Threads that are already waiting for objects in the claim method, will
159    * also wake up and receive an {@link IllegalStateException}.
160    *
161    * All objects that are already claimed when this method is called,
162    * will continue to function until they are
163    * {@link Poolable#release() released}.
164    *
165    * The shut down process is guaranteed to never deallocate objects that are
166    * currently claimed. Their deallocation will wait until they are released.
167    *
168    * @return A {@link Completion} instance that represents the shut down
169    * process.
170    */
171   public abstract Completion shutdown();
172 
173   /**
174    * Set the target size for this pool. The pool will strive to keep this many
175    * objects allocated at any one time.
176    *
177    * If the new target size is greater than the old one, the pool will allocate
178    * more objects until it reaches the target size. If, on the other hand, the
179    * new target size is less than the old one, the pool will deallocate more
180    * and allocate less, until the new target size is reached.
181    *
182    * No guarantees are made about when the pool actually reaches the target
183    * size. In fact, it may never happen as the target size can be changed as
184    * often as one sees fit.
185    *
186    * Pools that do not support a size less than 1 (which would deviate from the
187    * standard configuration space) will throw an
188    * {@link IllegalArgumentException} if passed 0 or less.
189    *
190    * Pools that do not support online resizing will throw an
191    * {@link UnsupportedOperationException}.
192    *
193    * @param size The new target size of the pool
194    */
195   public abstract void setTargetSize(int size);
196 
197   /**
198    * Get the currently configured target size of the pool. Note that this is
199    * _not_ the number of objects currently allocated by the pool - only
200    * the number of allocations the pool strives to keep alive.
201    *
202    * @return The current target size of this pool.
203    */
204   public abstract int getTargetSize();
205 
206   /**
207    * Get the {@link ManagedPool} instance that represents this pool.
208    * @return The {@link ManagedPool} instance for this pool.
209    */
210   public abstract ManagedPool getManagedPool();
211 
212   /**
213    * Get a thread-safe {@link PoolTap} implementation for this pool, which can
214    * be freely shared among multiple threads.
215    * @return A thread-safe {@link PoolTap}.
216    */
217   public final PoolTap<T> getThreadSafeTap() {
218     // We use the anonymous inner class to enforce encapsulation, which is
219     // probably the only reason anyone would wan to use this.
220     return new PoolTap<>() {
221       @Override
222       public T claim(Timeout timeout) throws PoolException, InterruptedException {
223         return Pool.this.claim(timeout);
224       }
225     };
226   }
227 
228   /**
229    * Get a {@link PoolTap} that only support access by one thread at a time.
230    * In other words, where {@link #claim(Timeout)} cannot be called
231    * concurrently in multiple threads *on the same tap*.
232    *
233    * The pool itself will still be thread-safe, but each thread that wishes to
234    * access the pool via a thread-local tap, must have their own tap instance.
235    *
236    * It is a use error to access these pool taps concurrently from multiple
237    * threads, or to transfer them from one thread to another without safe
238    * publication.
239    *
240    * @return A thread-locao {@link PoolTap}.
241    */
242   public abstract PoolTap<T> getThreadLocalTap();
243 }
