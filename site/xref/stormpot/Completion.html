16  package stormpot;
17  
18  /**
19   * A Completion represents some task that is going to be completed at some
20   * point in the future, or maybe already has completed. It is similar to
21   * {@link java.util.concurrent.Future Future} but without any options for
22   * cancellation or returning a result. Indeed, you cannot even precisely tell
23   * if the task has already completed, but the await methods will return
24   * immediately if that is the case.
25   * @author Chris Vest <mr.chrisvest@gmail.com>
26   * @see Pool#shutdown()
27   */
28  public interface Completion {
29    
30    /**
31     * Causes the current thread to wait until the completion is finished,
32     * or the thread is {@link Thread#interrupt() interrupted}, or the specified
33     * waiting time elapses.
34     *
35     * If the task represented by this completion has already completed,
36     * the method immediately returns `true`.
37     *
38     * If the current thread already has its interrupted status set upon entry
39     * to this method, or the thread is interrupted while waiting, then an
40     * {@link InterruptedException} is thrown and the current threads interrupted
41     * status is cleared.
42     *
43     * If the specified waiting time elapses, then the method returns `false`.
44     *
45     * @param timeout The timeout delimiting the maximum time to wait for the
46     * task to complete. Timeouts with zero or negative values will cause the
47     * method to return immediately.
48     * @return `true` if the task represented by this completion
49     * completed within the specified waiting time, or was already complete upon
50     * entry to this method; or `false` if the specified Timeout
51     * elapsed before the task could finished.
52     * @throws InterruptedException if the current thread is interrupted while
53     * waiting.
54     * @throws IllegalArgumentException if the provided `timeout` parameter is
55     * `null`.
56     */
57    boolean await(Timeout timeout) throws InterruptedException;
58  }
