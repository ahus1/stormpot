18  import java.util.Objects;
19  import java.util.concurrent.LinkedTransferQueue;
20  import java.util.concurrent.TimeUnit;

22  /**
23   * BlazePool is a highly optimised {@link Pool} implementation that consists of
24   * a queues of Poolable instances, the access to which is made faster with
25   * clever use of ThreadLocals.
26   *
27   * Object allocation always happens in a dedicated thread, off-loading the
28   * cost of allocating the pooled objects. This leads to reduced deviation
29   * in the times it takes claim method to complete, provided the pool is not
30   * depleted.
31   *
32   * BlazePool optimises for the case where the same threads need to claim and
33   * release objects over and over again. On the other hand, if the releasing
34   * thread tends to differ from the claiming thread, then the major optimisation
35   * in BlazePool is defeated, and performance regresses to a slow-path that is
36   * limited by contention on a blocking queue.
37   *
38   * @author Chris Vest <mr.chrisvest@gmail.com>
39   * @param <T> The type of {@link Poolable} managed by this pool.
40   */
41  final class BlazePool<T extends Poolable>
42      extends Pool<T> implements ManagedPool {
43  
44    private static final Exception SHUTDOWN_POISON =
45        new Exception("Stormpot Poison: Shutdown");
46    static final Exception EXPLICIT_EXPIRE_POISON =
47        new Exception("Stormpot Poison: Expired");
48  
49    private final LinkedTransferQueue<BSlot<T>> live;
50    private final RefillPile<T> disregardPile;
51    private final RefillPile<T> newAllocations;
52    private final AllocationController<T> allocator;
53    private final ThreadLocal<BSlotCache<T>> tlr;
54    private final Expiration<? super T> deallocRule;
55    private final MetricsRecorder metricsRecorder;
56  
57    /**
58     * A special slot used to signal that the pool has been shut down.
59     */
60    private final BSlot<T> poisonPill;
61  
62    private volatile boolean shutdown;
63  
64    /**
65     * Construct a new BlazePool instance based on the given {@link PoolBuilder}.
66     * @param builder The pool configuration to use.
67     */
68    BlazePool(PoolBuilder<T> builder, AllocationProcess factory) {
69      live = new LinkedTransferQueue<>();
70      disregardPile = new RefillPile<>(live);
71      newAllocations = new RefillPile<>(live);
72      tlr = new ThreadLocalBSlotCache<>();
73      poisonPill = new BSlot<>(live, null);
74      poisonPill.poison = SHUTDOWN_POISON;
75      deallocRule = builder.getExpiration();
76      metricsRecorder = builder.getMetricsRecorder();
77      allocator = factory.buildAllocationController(
78          live, disregardPile, newAllocations, builder, poisonPill);
79    }

81    @Override
82    public T claim(Timeout timeout)
83        throws PoolException, InterruptedException {
84      return tlrClaim(timeout, tlr.get());
85    }
86  
87    T tlrClaim(Timeout timeout, BSlotCache<T> cache)
88        throws PoolException, InterruptedException {
89      Objects.requireNonNull(timeout, "Timeout cannot be null.");
90      BSlot<T> slot = cache.slot;
91      // Note that the TLR slot at this point might have been tried by another
92      // thread, found to be expired, put on the dead-queue and deallocated.
93      // We handle this because slots always transition to the dead state before
94      // they are put on the dead-queue, and if they are dead, then the
95      // slot.live2claimTlr() call will fail.
96      // Then we will eventually find another slot from the live-queue that we
97      // can claim and make our new TLR slot.
98      if (slot != null && slot.live2claimTlr()) {
99        // Attempt the claim before checking the validity, because we might
100       // already have claimed it.
101       // If we checked validity before claiming, then we might find that it
102       // had expired, and throw it in the dead queue, causing a claimed
103       // Poolable to be deallocated before it is released.
104       if (!isInvalid(slot, cache, true)) {
105         slot.incrementClaims();
106         return slot.obj;
107       }
108       // We managed to tlr-claim the slot, but it turned out to be no good.
109       // That means we now have to transition it from tlr-claimed to dead.
110       // However, since we didn't pull it off of the live-queue, it might still
111       // be in the live-queue. And since it might be in the live-queue, it
112       // can't be put on the dead-queue. And since it can't be put on the
113       // dead-queue, it also cannot transition to the dead state.
114       // This effectively means that we have to transition it back to the live
115       // state, and then let some pull it off of the live-queue, check it
116       // again, and only then put it on the dead-queue.
117       // It's cumbersome, but we have to do it this way, in order to prevent
118       // duplicate entries in the queues. Otherwise we'd have a nasty memory
119       // leak on our hands.
120     }
121     // The thread-local claim failed, so we have to go through the slow-path.
122     return slowClaim(timeout, cache);
123   }
124 
125   private T slowClaim(Timeout timeout, BSlotCache<T> cache)
126       throws PoolException, InterruptedException {
127     // The slow-path for claim is in its own method to allow the fast-path to
128     // inline separately. At this point, taking a performance hit is
129     // inevitable anyway, so we're allowed a bit more leeway.
130     BSlot<T> slot;
131     long startNanos = NanoClock.nanoTime();
132     long timeoutNanos = timeout.getTimeoutInBaseUnit();
133     long timeoutLeft = timeoutNanos;
134     TimeUnit baseUnit = timeout.getBaseUnit();
135     long maxWaitQuantum = baseUnit.convert(100, TimeUnit.MILLISECONDS);
136     for (;;) {
137       slot = newAllocations.pop();
138       if (slot == null) {
139         long pollWait = Math.min(timeoutLeft, maxWaitQuantum);
140         slot = live.poll(pollWait, baseUnit);
141       }
142       if (slot == null) {
143         if (timeoutLeft <= 0) {
144           // We timed out while taking from the queue - just return null
145           return null;
146         } else {
147           timeoutLeft = NanoClock.timeoutLeft(startNanos, timeoutNanos);
148           disregardPile.refill();
149           continue;
150         }
151       }
152 
153       if (slot.live2claim()) {
154         if (isInvalid(slot, cache, false)) {
155           timeoutLeft = NanoClock.timeoutLeft(startNanos, timeoutNanos);
156           if (timeoutLeft <= 0) {
157             // There is no time left to poll the queue again - just return null
158             return null;
159           }
160         } else {
161           break;
162         }
163       } else {
164         disregardPile.push(slot);
165       }
166     }
167     slot.incrementClaims();
168     cache.slot = slot;
169     return slot.obj;
170   }

172   private boolean isInvalid(BSlot<T> slot, BSlotCache<T> cache, boolean isTlr) {
173     if (isUncommonlyInvalid(slot)) {
174       return handleUncommonInvalidation(slot, cache, isTlr);
175     }
176 
177     try {
178       return deallocRule.hasExpired(slot)
179           && handleCommonInvalidation(slot, cache, null);
180     } catch (Throwable ex) {
181       return handleCommonInvalidation(slot, cache, ex);
182     }
183   }
184 
185   private boolean isUncommonlyInvalid(BSlot<T> slot) {
186     return shutdown | slot.poison != null;
187   }
188 
189   private boolean handleUncommonInvalidation(
190       BSlot<T> slot, BSlotCache<T> cache, boolean isTlr) {
191     Exception poison = slot.poison;
192     if (poison != null) {
193       return dealWithSlotPoison(slot, cache, isTlr, poison);
194     } else {
195       kill(slot, cache);
196       throw new IllegalStateException("Pool has been shut down");
197     }
198   }
199 
200   private boolean handleCommonInvalidation(
201       BSlot<T> slot, BSlotCache<T> cache, Throwable exception) {
202     kill(slot, cache);
203     if (exception != null) {
204       String msg = "Got exception when checking whether an object had expired";
205       throw new PoolException(msg, exception);
206     }
207     return true;
208   }
209 
210   private boolean dealWithSlotPoison(
211       BSlot<T> slot, BSlotCache<T> cache, boolean isTlr, Exception poison) {
212     if (poison == SHUTDOWN_POISON) {
213       // The poison pill means the pool has been shut down. The pill was
214       // transitioned from live to claimed just prior to this check, so we
215       // must transition it back to live and put it back into the live-queue
216       // before throwing our exception.
217       // Because we always throw when we see it, it will never become a
218       // tlr-slot, and so we don't need to worry about transitioning from
219       // tlr-claimed to live.
220       slot.claim2live();
221       live.offer(poisonPill);
222       throw new IllegalStateException("Pool has been shut down");
223     } else {
224       kill(slot, cache);
225       if (isTlr || poison == EXPLICIT_EXPIRE_POISON) {
226         return true;
227       } else {
228         throw new PoolException("Allocation failed", poison);
229       }
230     }
231   }
232 
233   private void kill(BSlot<T> slot, BSlotCache<T> cache) {
234     // The use of claim2dead() here ensures that we don't put slots into the
235     // dead-queue more than once. Many threads might have this as their
236     // TLR-slot and try to tlr-claim it, but only when a slot has been normally
237     // claimed, that is, pulled off the live-queue, can it be put into the
238     // dead-queue. This helps ensure that a slot will only ever be in at most
239     // one queue.
240     if (slot.isClaimed()) {
241       slot.claim2dead();
242       allocator.offerDeadSlot(slot);
243     } else {
244       slot.claimTlr2live();
245       cache.slot = null;
246     }
247   }

249   @Override
250   public Completion shutdown() {
251     shutdown = true;
252     return allocator.shutdown();
253   }
254 
255   @Override
256   public void setTargetSize(int size) {
257     if (size < 0) {
258       throw new IllegalArgumentException(
259           "Target pool size must be positive");
260     }
261     if (shutdown) {
262       return;
263     }
264     allocator.setTargetSize(size);
265   }
